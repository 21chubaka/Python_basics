        function createInLinePlayer(mediaType, videoId, seotitle, heroImage){

            }

        function createYoutubePlayer(videoId){
            $('#video-player-container').hide();
            $('#video-player-container').html('<iframe style="width: 1024px; height: 576px" class="embed-responsive-item" src="https://www.youtube.com/embed/'+ videoId + '?modestbranding=1&amp;version=3&amp;hl=en_US" frameborder="0" allowfullscreen></iframe>');
            $('#video-player-container').show();
        }

        function createDivaPlayer(videoId, seotitle, heroImage){
            $('#video-player-container').hide();

            var postData = {
                autoplay: false,
                videoId: videoId,
                useClosedCaption: false,
                useHTML: true,
                playerPath: window.__env.divaPlayerUrl,
                settingsPath: window.__env.divaSettingsFile
            };

            $.support.cors = true;
            $.ajax({
                type: 'POST',
                url: window.__env.divaTokenizationUrl,
                data: JSON.stringify(postData),
                crossDomain: true,
                dataType: 'json',
                cache: false,
                async: false,
                headers: { "Content-Type": "application/json" },
                success: function (result) {
                    $('#video-player-container').html(result);
                    $('#divaIFrame').width(1024).height(576);
                    $('#video-player-container').show();
                    //resize();
                },
                error: function (result) {
                    return false;
                }
            });

            window.addEventListener('resize', function () {
                resize();
            });

            function resize() {
                var iframe = document.getElementById('divaIFrame');
                var videoContainer = document.getElementById('videoContainer');

                var h = 0;
                var w = 0;

                if (window.innerWidth > 768) {
                    h = window.innerHeight * 80 / 100;//videoContainer.clientHeight;
                    w = window.innerWidth * 80 / 100; //videoContainer.clientWidth - 50;

                } else {
                    h = window.innerHeight;
                    w = window.innerWidth;
                }

                var dh = Math.round(w * 9 / 16);
                var dw = Math.round(16 * h / 9);

                if (dw > w)
                    dw = w;
                if (dh > h)
                    dh = h;

                dw = dw - (dw % 16);
                dh = dh - (dh % 9);

                iframe.style.height = '100%' // dh + 'px';
                iframe.style.width = '100%' //dw + 'px';
            }
        }
